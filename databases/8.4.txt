create table users (
  id integer primary key,
  first varchar(255),
  last varchar(255)
);

create table reviews (
  id integer primary key,
  stars INT,
  comment varchar(255),
  business_id int,
  user_id int,
  FOREIGN KEY (business_id) references businesses(id),
  FOREIGN KEY(user_id) references users(id)
);

CREATE TABLE businesses (
  id integer primary key,
  name varchar(255)
);
